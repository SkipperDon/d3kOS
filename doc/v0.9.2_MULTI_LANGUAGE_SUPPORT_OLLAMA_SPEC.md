## MISSION STATEMENT

Add full 8-language support to d3kOS. Every piece of user-visible text — dashboard labels,
voice responses, settings, onboarding wizard, alerts — must display in the user's chosen language.

**Supported Languages:**
| Code | Language | Region | Voice TTS Model |
|------|----------|--------|-----------------|
| en | English | Default | piper-en-us (existing) |
| fr | French | Canada/France | piper-fr-fr |
| de | German | Germany/Austria | piper-de-de |
| es | Spanish | Spain/Latin America | piper-es-es |
| pt | Portuguese | Portugal/Brazil | piper-pt-br |
| nl | Dutch | Netherlands | piper-nl-nl |
| sv | Swedish | Sweden | piper-sv-se |
| ja | Japanese | Japan | piper-ja-jp |

**Approach:**
- All UI strings stored in JSON translation files
- `i18n.js` JavaScript library handles browser-side translation
- Backend Python services load translation for voice responses
- Language stored in `user-preferences.json` (extends existing API)
- Voice TTS models downloaded on demand when language selected

---

## ARCHITECTURE

```
Browser
  ├── i18n.js (translation library)
  │     ├── load /i18n/{lang}.json on init
  │     ├── translate all [data-i18n] elements
  │     └── listen for language change events
  │
  ├── All HTML pages: add data-i18n="key" to text elements
  └── settings.html: language selector dropdown

  /i18n/{lang}.json  ← static translation files
  (en, fr, de, es, pt, nl, sv, ja)

Backend:
  preferences-api.py (port 8107)
    POST /api/preferences  {"language": "fr"}

  query_handler.py (voice responses)
    reads language from user-preferences.json
    loads /opt/d3kos/config/i18n/{lang}.json for responses

  piper TTS
    uses language-specific model file
    /opt/d3kos/models/voice/piper-{lang}/
```

## PHASE 3: i18n.JS LIBRARY

**File:** `/var/www/html/js/i18n.js`

```javascript
/**
 * d3kOS i18n — Internationalization Library
 * Version: 0.15.0
 * Translates [data-i18n] elements and provides t() function
 */

const i18n = {
  _strings: {},
  _lang: 'en',

  SUPPORTED: ['en', 'fr', 'de', 'es', 'pt', 'nl', 'sv', 'ja'],
  LANGUAGE_NAMES: {
    en: 'English', fr: 'Français', de: 'Deutsch', es: 'Español',
    pt: 'Português', nl: 'Nederlands', sv: 'Svenska', ja: '日本語'
  },
  DIRECTIONS: {
    en: 'ltr', fr: 'ltr', de: 'ltr', es: 'ltr',
    pt: 'ltr', nl: 'ltr', sv: 'ltr', ja: 'ltr'
  },

  async init() {
    // Load language from backend preference
    try {
      const r = await fetch('/api/preferences');
      const data = await r.json();
      this._lang = data.language || 'en';
    } catch(e) {
      this._lang = localStorage.getItem('d3kos-language') || 'en';
    }
    if (!this.SUPPORTED.includes(this._lang)) this._lang = 'en';
    await this.loadLanguage(this._lang);
    this.translatePage();
    document.documentElement.lang = this._lang;
    document.documentElement.dir = this.DIRECTIONS[this._lang] || 'ltr';
  },

  async loadLanguage(lang) {
    try {
      const r = await fetch(`/i18n/${lang}.json`);
      if (!r.ok) throw new Error('Not found');
      const data = await r.json();
      this._strings = this._flatten(data);
      this._lang = lang;
      localStorage.setItem('d3kos-language', lang);
    } catch(e) {
      if (lang !== 'en') {
        console.warn(`i18n: Could not load ${lang}, falling back to English`);
        await this.loadLanguage('en');
      }
    }
  },

  _flatten(obj, prefix = '') {
    const result = {};
    for (const [key, val] of Object.entries(obj)) {
      if (key === '_meta') continue;
      const full = prefix ? `${prefix}.${key}` : key;
      if (typeof val === 'object' && val !== null && !Array.isArray(val)) {
        Object.assign(result, this._flatten(val, full));
      } else {
        result[full] = val;
      }
    }
    return result;
  },

  t(key, vars = {}) {
    let str = this._strings[key] || key;
    // Replace {variable} placeholders
    for (const [k, v] of Object.entries(vars)) {
      str = str.replace(new RegExp(`\\{${k}\\}`, 'g'), v);
    }
    return str;
  },

  translatePage() {
    // Translate all [data-i18n] attributes
    document.querySelectorAll('[data-i18n]').forEach(el => {
      const key = el.getAttribute('data-i18n');
      const translated = this.t(key);
      if (translated !== key) {
        if (el.tagName === 'INPUT' && el.placeholder !== undefined) {
          el.placeholder = translated;
        } else if (el.tagName === 'INPUT' || el.tagName === 'BUTTON') {
          el.value = translated;
        } else {
          el.textContent = translated;
        }
      }
    });
    // Translate [data-i18n-placeholder]
    document.querySelectorAll('[data-i18n-placeholder]').forEach(el => {
      el.placeholder = this.t(el.getAttribute('data-i18n-placeholder'));
    });
    // Translate [data-i18n-title]
    document.querySelectorAll('[data-i18n-title]').forEach(el => {
      el.title = this.t(el.getAttribute('data-i18n-title'));
    });
  },

  async setLanguage(lang) {
    if (!this.SUPPORTED.includes(lang)) return;
    await this.loadLanguage(lang);
    this.translatePage();
    // Save to backend
    await fetch('/api/preferences', {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({ language: lang })
    }).catch(() => {});
    window.dispatchEvent(new CustomEvent('languageChanged', { detail: { lang } }));
    document.documentElement.lang = lang;
  },

  getCurrentLanguage() { return this._lang; },
  getLanguageName(code) { return this.LANGUAGE_NAMES[code] || code; }
};

// Auto-init when DOM is ready
if (document.readyState === 'loading') {
  document.addEventListener('DOMContentLoaded', () => i18n.init());
} else {
  i18n.init();
}
```

## PHASE 6: VOICE ASSISTANT — MULTI-LANGUAGE

**File:** `/opt/d3kos/services/ai/query_handler.py`

Add language-aware response formatting:

```python
import json

def _load_i18n(self, lang: str = None) -> dict:
    """Load translation strings for voice responses."""
    if not lang:
        try:
            with open('/opt/d3kos/config/user-preferences.json') as f:
                prefs = json.load(f)
            lang = prefs.get('language', 'en')
        except Exception:
            lang = 'en'
    i18n_path = f'/var/www/html/i18n/{lang}.json'
    try:
        with open(i18n_path) as f:
            data = json.load(f)
        # Flatten the nested dict
        return self._flatten_i18n(data)
    except Exception:
        return {}

def _flatten_i18n(self, obj, prefix=''):
    result = {}
    for k, v in obj.items():
        if k == '_meta': continue
        full = f'{prefix}.{k}' if prefix else k
        if isinstance(v, dict):
            result.update(self._flatten_i18n(v, full))
        else:
            result[full] = v
    return result

def _t(self, key: str, strings: dict, **vars) -> str:
    """Get translated string with variable substitution."""
    s = strings.get(key, key)
    for k, v in vars.items():
        s = s.replace('{' + k + '}', str(v))
    return s

def format_voice_response(self, response_type: str, value=None) -> str:
    strings = self._load_i18n()
    key = f'voice.{response_type}_response'
    if value is not None:
        return self._t(key, strings, value=value)
    return self._t(key, strings)
```
