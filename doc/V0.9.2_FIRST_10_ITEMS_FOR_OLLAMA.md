## Item 1: Temperature Conversion (°F ↔ °C)

### Specification
Create JavaScript function to convert between Fahrenheit and Celsius.

### Implementation Requirements
```javascript
/**
 * Convert Fahrenheit to Celsius
 * Formula: (°F - 32) × 5/9 = °C
 * @param {number} fahrenheit - Temperature in Fahrenheit
 * @returns {number} Temperature in Celsius (rounded to 1 decimal)
 * @example temperatureToC(185) => 85.0
 */
function temperatureToC(fahrenheit) {
    if (typeof fahrenheit !== 'number' || isNaN(fahrenheit)) {
        throw new Error('Invalid input: must be a number');
    }
    return Math.round(((fahrenheit - 32) * 5/9) * 10) / 10;
}

/**
 * Convert Celsius to Fahrenheit
 * Formula: (°C × 9/5) + 32 = °F
 * @param {number} celsius - Temperature in Celsius
 * @returns {number} Temperature in Fahrenheit (rounded to 1 decimal)
 * @example temperatureToF(85) => 185.0
 */
function temperatureToF(celsius) {
    if (typeof celsius !== 'number' || isNaN(celsius)) {
        throw new Error('Invalid input: must be a number');
    }
    return Math.round(((celsius * 9/5) + 32) * 10) / 10;
}
```

## Item 3: Speed Conversion (knots+MPH ↔ knots+km/h)

### Specification
Convert speed display format between:
- Imperial: "10.0 kts (11.5 mph)"
- Metric: "10.0 kts (18.5 km/h)"

Note: Knots always displayed, secondary unit varies.

### Implementation Requirements
```javascript
/**
 * Convert knots to km/h
 * Formula: knots × 1.852 = km/h
 * @param {number} knots - Speed in knots
 * @returns {number} Speed in km/h (rounded to 1 decimal)
 * @example knotsToKmh(10) => 18.5
 */
function knotsToKmh(knots) {
    if (typeof knots !== 'number' || isNaN(knots)) {
        throw new Error('Invalid input: must be a number');
    }
    return Math.round(knots * 1.852 * 10) / 10;
}

/**
 * Convert knots to mph
 * Formula: knots × 1.15078 = mph
 * @param {number} knots - Speed in knots
 * @returns {number} Speed in mph (rounded to 1 decimal)
 * @example knotsToMph(10) => 11.5
 */
function knotsToMph(knots) {
    if (typeof knots !== 'number' || isNaN(knots)) {
        throw new Error('Invalid input: must be a number');
    }
    return Math.round(knots * 1.15078 * 10) / 10;
}

/**
 * Format speed display based on unit preference
 * @param {number} knots - Speed in knots
 * @param {string} system - 'imperial' or 'metric'
 * @returns {string} Formatted speed string
 * @example formatSpeed(10, 'imperial') => "10.0 kts (11.5 mph)"
 */
function formatSpeed(knots, system) {
    if (system === 'metric') {
        return `${knots.toFixed(1)} kts (${knotsToKmh(knots)} km/h)`;
    } else {
        return `${knots.toFixed(1)} kts (${knotsToMph(knots)} mph)`;
    }
}
```

### Test Cases
