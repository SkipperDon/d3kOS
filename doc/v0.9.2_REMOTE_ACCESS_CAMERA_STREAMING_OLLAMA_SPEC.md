**SCOPE — what this feature delivers:**
1. **Remote Dashboard API** — read-only vessel data via authenticated API (mobile app ready)
2. **Camera Stream Relay** — RTSP → HLS conversion for remote viewing in browser/mobile
3. **Remote Alerts** — push boat alerts to AtMyBoat.com cloud endpoint
4. **Remote Diagnostics** — query system health from anywhere

**OUT OF SCOPE for this spec:**
- Mobile app (iOS/Android) — separate spec
- Full fleet management — v0.10.x
- AtMyBoat.com cloud infrastructure — owner provides endpoint URL

**Security:** All remote endpoints are token-authenticated.
Token stored in `/opt/d3kos/config/remote-access.json`.

## ARCHITECTURE

```
Internet/Mobile
      │ HTTPS + Bearer token
      ▼
AtMyBoat.com relay (cloud)
      │ polls or webhooks
      ▼
Pi Remote API (port 8096)
      ├── GET  /remote/vessel/status     → all live data
      ├── GET  /remote/vessel/engine     → engine snapshot
      ├── GET  /remote/vessel/location   → GPS position
      ├── GET  /remote/alerts/recent     → last 50 alerts
      ├── GET  /remote/system/health     → Pi system health
      ├── POST /remote/alerts/acknowledge → ack alert
      └── GET  /remote/camera/snapshot   → latest frame JPEG

HLS Stream Service (port 8097)
      ├── Converts RTSP → HLS segments
      ├── GET /hls/{camera_id}/stream.m3u8
      └── GET /hls/{camera_id}/seg_*.ts

Nginx:
      ├── /remote/* → port 8096
      └── /hls/*    → port 8097
```

## PHASE 2: REMOTE DATA API

**File:** `/opt/d3kos/services/remote/remote-api.py`

```python
#!/usr/bin/env python3
"""
d3kOS Remote Access API
Version: 0.9.6
Port: 8096
Provides authenticated read-only access to vessel data for remote monitoring.
"""
import json
import os
import subprocess
import time
from datetime import datetime, timedelta
from functools import wraps

import requests
from flask import Flask, jsonify, request, send_file, abort
from flask_cors import CORS

app = Flask(__name__)
CORS(app)

REMOTE_CONFIG_FILE = '/opt/d3kos/config/remote-access.json'
PREFS_FILE = '/opt/d3kos/config/user-preferences.json'
ALERTS_FILE = '/opt/d3kos/data/alerts.json'
FW_RESULTS_FILE = '/tmp/d3kos_forward_watch.json'
CAMERA_FRAME_FILE = '/tmp/d3kos_frame_cam01.jpg'
SIGNALK_URL = 'http://localhost:3000'


def load_config():
    try:
        with open(REMOTE_CONFIG_FILE) as f:
            return json.load(f)
    except Exception:
        return {'enabled': False, 'api_token': ''}


def token_required(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        cfg = load_config()
        if not cfg.get('enabled', False):
            abort(503, 'Remote access not enabled')
        token = request.headers.get('Authorization', '').replace('Bearer ', '').strip()
        if not token or token != cfg.get('api_token', ''):
            abort(401, 'Invalid or missing token')
        return f(*args, **kwargs)
    return decorated


def get_signalk_value(path: str):
    """Fetch a value from SignalK."""
    try:
        r = requests.get(
            f'{SIGNALK_URL}/signalk/v1/api/vessels/self/{path.replace(".", "/")}',
            timeout=2
        )
        if r.status_code == 200:
            return r.json().get('value')
    except Exception:
        pass
    return None


def get_vessel_snapshot():
    """Get current vessel data from SignalK."""
    try:
        r = requests.get(f'{SIGNALK_URL}/signalk/v1/api/vessels/self', timeout=3)
        if r.status_code == 200:
            raw = r.json()
            # Extract key values
            prop = raw.get('propulsion', {}).get('0', {})
            nav = raw.get('navigation', {})
            tanks = raw.get('tanks', {})
            electrical = raw.get('electrical', {}).get('batteries', {}).get('0', {})

            return {
                'engine': {
                    'rpm': _val(prop, 'revolutions'),
                    'oil_pressure_psi': _val(prop, 'oilPressure'),
                    'coolant_temp_f': _val(prop, 'temperature'),
                    'hours': _val(prop, 'runTime'),
                    'state': _val(prop, 'state')
                },
                'navigation': {
                    'latitude': _nav_lat(nav),
                    'longitude': _nav_lon(nav),
                    'speed_knots': _val(nav, 'speedOverGround'),
                    'heading_true': _val(nav, 'headingTrue'),
                    'cog_true': _val(nav, 'courseOverGroundTrue')
                },
                'tanks': {
                    'fuel_pct': _val(tanks, 'fuel.0.currentLevel'),
                    'fresh_water_pct': _val(tanks, 'freshWater.0.currentLevel')
                },
                'electrical': {
                    'battery_voltage': _val(electrical, 'voltage')
                },
                'timestamp': datetime.now().isoformat()
            }
    except Exception as e:
        return {'error': str(e), 'timestamp': datetime.now().isoformat()}
    return {'timestamp': datetime.now().isoformat()}


def _val(d, key):
    """Safe nested dict access."""
    keys = key.split('.')
    for k in keys:
        if isinstance(d, dict):
            d = d.get(k, {})
        else:
            return None
    return d.get('value') if isinstance(d, dict) else d


def _nav_lat(nav):
    try:
        return nav.get('position', {}).get('value', {}).get('latitude')
    except Exception:
        return None


def _nav_lon(nav):
    try:
        return nav.get('position', {}).get('value', {}).get('longitude')
    except Exception:
        return None


def get_system_health():
    """Get Pi system health metrics."""
    try:
        cpu = float(subprocess.check_output(['top', '-bn1'], text=True)
                    .split('Cpu(s):')[1].split('id')[0].split(',')[-1].strip().replace('%', ''))
        cpu_used = round(100 - cpu, 1)
    except Exception:
        cpu_used = None

    try:
        mem_out = subprocess.check_output(['free', '-m'], text=True).split('\n')[1].split()
        mem_used_pct = round(int(mem_out[2]) / int(mem_out[1]) * 100, 1)
    except Exception:
        mem_used_pct = None

    try:
        disk_out = subprocess.check_output(['df', '-h', '/'], text=True).split('\n')[1].split()
        disk_used_pct = disk_out[4].replace('%', '')
    except Exception:
        disk_used_pct = None

    try:
        temp_out = subprocess.check_output(['vcgencmd', 'measure_temp'], text=True)
        cpu_temp_c = float(temp_out.replace("temp=", "").replace("'C\n", ""))
    except Exception:
        cpu_temp_c = None

    return {
        'cpu_usage_pct': cpu_used,
        'memory_used_pct': mem_used_pct,
        'disk_used_pct': disk_used_pct,
        'cpu_temp_c': cpu_temp_c,
        'uptime': subprocess.check_output(['uptime', '-p'], text=True).strip(),
        'timestamp': datetime.now().isoformat()
    }


def get_recent_alerts(limit=50):
    """Load recent alerts from alerts file."""
    try:
        with open(ALERTS_FILE) as f:
            alerts = json.load(f)
        return sorted(alerts, key=lambda a: a.get('timestamp', ''), reverse=True)[:limit]
    except Exception:
        return []


# --- API ENDPOINTS ---

@app.route('/remote/vessel/status', methods=['GET'])
@token_required
def vessel_status():
    snapshot = get_vessel_snapshot()
    try:
        with open(FW_RESULTS_FILE) as f:
            fw = json.load(f)
        snapshot['forward_watch'] = {
            'alert_level': fw.get('alert_level', 'NONE'),
            'detection_count': fw.get('detection_count', 0),
            'timestamp': fw.get('timestamp')
        }
    except Exception:
        snapshot['forward_watch'] = {'alert_level': 'UNKNOWN'}
    return jsonify(snapshot)


@app.route('/remote/vessel/engine', methods=['GET'])
@token_required
def vessel_engine():
    snapshot = get_vessel_snapshot()
    return jsonify({'engine': snapshot.get('engine', {}), 'timestamp': snapshot.get('timestamp')})


@app.route('/remote/vessel/location', methods=['GET'])
@token_required
def vessel_location():
    snapshot = get_vessel_snapshot()
    return jsonify({'navigation': snapshot.get('navigation', {}), 'timestamp': snapshot.get('timestamp')})


@app.route('/remote/system/health', methods=['GET'])
@token_required
def system_health():
    return jsonify(get_system_health())


@app.route('/remote/alerts/recent', methods=['GET'])
@token_required
def recent_alerts():
    limit = min(int(request.args.get('limit', 50)), 200)
    return jsonify({'alerts': get_recent_alerts(limit)})


@app.route('/remote/camera/snapshot', methods=['GET'])
@token_required
def camera_snapshot():
    cfg = load_config()
    if 'camera_snapshot' not in cfg.get('allowed_data', []):
        abort(403, 'Camera snapshots not enabled')
    # Request fresh frame from camera service
    try:
        r = requests.get('http://localhost:8084/cameras/cam01/frame', timeout=10)
        if r.status_code == 200:
            with open('/tmp/remote_snapshot.jpg', 'wb') as f:
                f.write(r.content)
            return send_file('/tmp/remote_snapshot.jpg', mimetype='image/jpeg')
    except Exception as e:
        abort(503, f'Camera unavailable: {str(e)}')


@app.route('/remote/alerts/acknowledge', methods=['POST'])
@token_required
def acknowledge_alert():
    data = request.get_json(force=True) or {}
    alert_id = data.get('alert_id')
    if not alert_id:
        return jsonify({'error': 'alert_id required'}), 400
    # TODO: Mark alert acknowledged in alerts.json
    return jsonify({'success': True, 'alert_id': alert_id, 'status': 'acknowledged'})


@app.route('/remote/config', methods=['GET'])
def get_remote_config():
    """Non-authenticated endpoint — returns non-sensitive config."""
    cfg = load_config()
    return jsonify({
        'enabled': cfg.get('enabled', False),
        'push_alerts': cfg.get('push_alerts', True),
        'camera_stream_enabled': cfg.get('camera_stream_enabled', False)
    })


@app.route('/remote/config', methods=['POST'])
def set_remote_config():
    """Settings page uses this to enable/configure remote access."""
    # Requires local access only (no token — only from Pi's localhost)
    if request.remote_addr not in ('127.0.0.1', '::1', '192.168.1.237'):
        abort(403, 'Config changes only allowed from local network')
    data = request.get_json(force=True) or {}
    cfg = load_config()
    allowed_keys = ['enabled', 'api_token', 'cloud_endpoint', 'push_alerts',
                    'camera_stream_enabled', 'hls_quality', 'allowed_data']
    for k in allowed_keys:
        if k in data:
            cfg[k] = data[k]
    os.makedirs(os.path.dirname(REMOTE_CONFIG_FILE), exist_ok=True)
    with open(REMOTE_CONFIG_FILE, 'w') as f:
        json.dump(cfg, f, indent=2)
    return jsonify({'success': True, 'config': cfg})


@app.route('/health', methods=['GET'])
def health():
    cfg = load_config()
    return jsonify({
        'status': 'ok',
        'service': 'remote-api',
        'version': '0.9.6',
        'remote_enabled': cfg.get('enabled', False),
        'camera_stream_enabled': cfg.get('camera_stream_enabled', False)
    })


if __name__ == '__main__':
    app.run(host='0.0.0.0', port=8096, debug=False)
```

## PHASE 3: HLS CAMERA STREAMING

**File:** `/opt/d3kos/services/remote/hls-stream.sh`

This script converts RTSP → HLS using ffmpeg.

```bash
#!/bin/bash
# d3kOS HLS Camera Stream Generator
# Version: 0.9.6
# Converts RTSP camera feed to HLS for remote viewing

CAMERA_ID="${1:-cam01}"
RTSP_URL="${2:-rtsp://admin:d3kos2026@10.42.0.100:554/h264Preview_01_sub}"
HLS_DIR="/var/www/html/hls/${CAMERA_ID}"
SEGMENT_DURATION=2
SEGMENTS_KEPT=5

mkdir -p "${HLS_DIR}"

echo "Starting HLS stream for ${CAMERA_ID}"
echo "RTSP: ${RTSP_URL}"
echo "Output: ${HLS_DIR}/stream.m3u8"

ffmpeg -y \
  -rtsp_transport tcp \
  -i "${RTSP_URL}" \
  -c:v copy \
  -an \
  -f hls \
  -hls_time ${SEGMENT_DURATION} \
  -hls_list_size ${SEGMENTS_KEPT} \
  -hls_flags delete_segments+append_list \
  -hls_segment_filename "${HLS_DIR}/seg_%03d.ts" \
  "${HLS_DIR}/stream.m3u8" \
  2>&1

echo "HLS stream ended for ${CAMERA_ID}"
```


**Nginx configuration for HLS:** Add to `/etc/nginx/sites-enabled/default`:

```nginx
# Remote API
location /remote/ {
    proxy_pass http://127.0.0.1:8096/remote/;
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_read_timeout 15s;
    # Security: only allow from local network or with valid Bearer token
}

# HLS Camera Streams (static files served by nginx)
location /hls/ {
    root /var/www/html;
    add_header Cache-Control "no-cache";
    add_header Access-Control-Allow-Origin "*";
    types {
        application/vnd.apple.mpegurl m3u8;
        video/mp2t ts;
    }
}
```
