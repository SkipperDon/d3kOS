## MISSION STATEMENT

Expand d3kOS from single-camera to multi-camera (up to 4+) with:
1. **Camera Registry** — JSON config defining all cameras
2. **Multi-Camera API** — per-camera endpoints (port 8084)
3. **Single View with Switcher** — select which camera to view full-screen
4. **Grid View** — see all cameras simultaneously (2×2 layout)
5. **DHCP Reservations** — static IPs for all cameras

Forward Watch (AI detection on bow camera) is in a separate spec:
`v0.9.2_FORWARD_WATCH_OBSTACLE_AVOIDANCE_OLLAMA_SPEC.md`

```python
#!/usr/bin/env python3
"""
d3kOS Multi-Camera Stream Manager
Version: 0.9.3
Port: 8084
Manages multiple IP cameras with per-camera API endpoints
"""
import json
import os
import subprocess
import threading
import time
import tempfile
from datetime import datetime
from typing import Dict, Optional

from flask import Flask, jsonify, send_file, request
from flask_cors import CORS

app = Flask(__name__)
CORS(app)

CAMERAS_CONFIG = '/opt/d3kos/config/cameras.json'
RECORDINGS_BASE = '/media/d3kos/recordings'

# Active camera streams: {camera_id: {'process': Process, 'frame_path': str}}
active_streams: Dict[str, dict] = {}
active_camera_id: Optional[str] = None
stream_lock = threading.Lock()


def load_cameras() -> dict:
    try:
        with open(CAMERAS_CONFIG) as f:
            return json.load(f)
    except Exception as e:
        return {'cameras': [], 'settings': {}}


def get_camera(camera_id: str) -> Optional[dict]:
    cfg = load_cameras()
    for cam in cfg['cameras']:
        if cam['id'] == camera_id:
            return cam
    return None


def get_enabled_cameras() -> list:
    cfg = load_cameras()
    return [c for c in cfg['cameras'] if c.get('enabled', False)]


def test_camera_reachable(ip: str, timeout: int = 3) -> bool:
    result = subprocess.run(
        ['ping', '-c', '1', '-W', str(timeout), ip],
        capture_output=True
    )
    return result.returncode == 0


def capture_frame(camera_id: str, stream_type: str = 'main') -> Optional[str]:
    """Capture a single JPEG frame from camera using ffmpeg."""
    cam = get_camera(camera_id)
    if not cam:
        return None
    rtsp_url = cam['rtsp'].get(stream_type, cam['rtsp']['main'])
    output_path = f'/tmp/d3kos_frame_{camera_id}.jpg'
    try:
        result = subprocess.run(
            ['ffmpeg', '-y', '-rtsp_transport', 'tcp',
             '-i', rtsp_url, '-frames:v', '1', '-q:v', '3', output_path],
            capture_output=True, timeout=10
        )
        if result.returncode == 0 and os.path.exists(output_path):
            return output_path
    except (subprocess.TimeoutExpired, Exception):
        pass
    return None


# --- API ENDPOINTS ---

@app.route('/cameras/list', methods=['GET'])
def list_cameras():
    cfg = load_cameras()
    cameras_status = []
    for cam in cfg['cameras']:
        online = test_camera_reachable(cam['ip'])
        cameras_status.append({
            'id': cam['id'],
            'name': cam['name'],
            'location': cam['location'],
            'purpose': cam['purpose'],
            'ip': cam['ip'],
            'enabled': cam.get('enabled', False),
            'online': online,
            'detection_enabled': cam.get('detection_enabled', False),
            'active': (cam['id'] == active_camera_id)
        })
    return jsonify({'cameras': cameras_status, 'active_camera': active_camera_id})


@app.route('/cameras/<camera_id>/status', methods=['GET'])
def camera_status(camera_id: str):
    cam = get_camera(camera_id)
    if not cam:
        return jsonify({'error': 'Camera not found'}), 404
    online = test_camera_reachable(cam['ip'])
    return jsonify({
        'id': cam['id'],
        'name': cam['name'],
        'ip': cam['ip'],
        'online': online,
        'enabled': cam.get('enabled', False),
        'active': (camera_id == active_camera_id),
        'rtsp_main': cam['rtsp']['main'],
        'rtsp_sub': cam['rtsp']['sub']
    })


@app.route('/cameras/<camera_id>/frame', methods=['GET'])
def camera_frame(camera_id: str):
    stream_type = request.args.get('stream', 'main')
    cam = get_camera(camera_id)
    if not cam:
        return jsonify({'error': 'Camera not found'}), 404
    frame_path = capture_frame(camera_id, stream_type)
    if frame_path and os.path.exists(frame_path):
        return send_file(frame_path, mimetype='image/jpeg')
    return jsonify({'error': 'Could not capture frame'}), 503


@app.route('/cameras/switch/<camera_id>', methods=['POST'])
def switch_camera(camera_id: str):
    global active_camera_id
    cam = get_camera(camera_id)
    if not cam:
        return jsonify({'error': 'Camera not found'}), 404
    with stream_lock:
        active_camera_id = camera_id
    return jsonify({'success': True, 'active_camera': camera_id, 'camera': cam['name']})


@app.route('/cameras/grid', methods=['GET'])
def camera_grid():
    """Return frames from all enabled cameras for grid view."""
    enabled = get_enabled_cameras()
    frames = {}
    for cam in enabled:
        frame_path = capture_frame(cam['id'], 'sub')
        frames[cam['id']] = {
            'name': cam['name'],
            'location': cam['location'],
            'online': (frame_path is not None),
            'frame_url': f'/cameras/{cam["id"]}/frame?stream=sub'
        }
    return jsonify({'grid': frames})


@app.route('/cameras/discover', methods=['POST'])
def discover_cameras():
    """Scan 10.42.0.100-109 for Reolink cameras."""
    found = []
    for i in range(100, 110):
        ip = f'10.42.0.{i}'
        if test_camera_reachable(ip, timeout=1):
            found.append({'ip': ip, 'discovered_at': datetime.now().isoformat()})
    return jsonify({'discovered': found, 'count': len(found)})


@app.route('/cameras/<camera_id>/enable', methods=['POST'])
def enable_camera(camera_id: str):
    cfg = load_cameras()
    for cam in cfg['cameras']:
        if cam['id'] == camera_id:
            cam['enabled'] = True
            with open(CAMERAS_CONFIG, 'w') as f:
                json.dump(cfg, f, indent=2)
            return jsonify({'success': True, 'camera': camera_id, 'enabled': True})
    return jsonify({'error': 'Camera not found'}), 404


@app.route('/cameras/<camera_id>/disable', methods=['POST'])
def disable_camera(camera_id: str):
    cfg = load_cameras()
    for cam in cfg['cameras']:
        if cam['id'] == camera_id:
            cam['enabled'] = False
            with open(CAMERAS_CONFIG, 'w') as f:
                json.dump(cfg, f, indent=2)
            return jsonify({'success': True, 'camera': camera_id, 'enabled': False})
    return jsonify({'error': 'Camera not found'}), 404


# Keep legacy single-camera endpoints for backwards compatibility
@app.route('/camera/status', methods=['GET'])
def camera_status_legacy():
    return camera_status('cam01')


@app.route('/camera/frame', methods=['GET'])
def camera_frame_legacy():
    return camera_frame('cam01')


@app.route('/health', methods=['GET'])
def health():
    enabled = get_enabled_cameras()
    return jsonify({
        'status': 'ok',
        'service': 'camera-stream-manager',
        'version': '0.9.3',
        'cameras_configured': len(load_cameras().get('cameras', [])),
        'cameras_enabled': len(enabled),
        'active_camera': active_camera_id
    })


if __name__ == '__main__':
    cfg = load_cameras()
    if cfg['cameras']:
        active_camera_id = cfg['settings'].get('default_camera', 'cam01')
    app.run(host='0.0.0.0', port=8084, debug=False)
```
